12.4.2 記述統計量のグラフ

df %>%
    group_by(Pref) %>%
    summarise(Score   = mean(Score, na.rm = TRUE),
              .groups = "drop") %>%
    ggplot() +
    geom_bar(aes(x = Pref, y = Score), stat = "identity") +
    coord_cartesian(ylim = c(1, 5)) +
    labs(x = "都府県", y = "口コミ評価の平均値 (1~5)") +
    theme_gray(base_family = "HiraKakuProN-W3")

df %>%
    group_by(Pref) %>%
    summarise(Score   = mean(Score, na.rm = TRUE),
              .groups = "drop") %>%
    ggplot() +
    geom_bar(aes(x = Pref, y = Score), stat = "identity") +
    labs(x = "都府県", y = "口コミ評価の平均値 (1~5)") +
    theme_gray(base_family = "HiraKakuProN-W3")

df %>%
    group_by(Pref) %>%
    summarise(Score   = mean(Score, na.rm = TRUE),
              .groups = "drop") %>%
    arrange(desc(Score)) %>%
    mutate(Pref = fct_inorder(Pref)) %>%
    ggplot() +
    geom_bar(aes(x = Pref, y = Score), stat = "identity") +
    coord_cartesian(ylim = c(1, 5)) +
    labs(x = "都府県", y = "口コミ評価の平均値 (1~5)") +
    theme_gray(base_family = "HiraKakuProN-W3")

df %>%
    group_by(Pref) %>%
    summarise(ScoreN  = sum(ScoreN),
              .groups = "drop") %>%
    arrange(desc(ScoreN)) %>%
    mutate(Pref = fct_inorder(Pref)) %>%
    ggplot() +
    geom_bar(aes(x = Pref, y = ScoreN), stat = "identity") +
    labs(x = "都府県", y = "口コミの合計") +
    theme_gray(base_family = "HiraKakuProN-W3")

    filter(!is.na(Walk)) %>%

## `geom_smooth()` using formula 'y ~ x'

df %>%
    filter(!is.na(Walk)) %>%
    mutate(EW = recode(Pref, .default = "関西",
                       "東京都" = "関東", "神奈川県" = "関東",
                       "千葉県" = "関東", "埼玉県"   = "関東")) %>%
    ggplot(aes(x = Walk, y = ScoreN, color = EW)) +
    geom_point() +
    geom_smooth(method = "loess") +

df %>%
    filter(!is.na(Budget), !is.na(Score)) %>%
    mutate(EW = recode(Pref, .default = "関西",
